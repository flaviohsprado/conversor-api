interface ITransactionError {
  code: number;
  type: string;
  info: string;
}

interface IQuery {
  from: string;
  to: string;
  amount: number;
}

interface IInfo {
  timestamp: number;
  rate: number;
}

export type ITransaction = {
  success: boolean;
} & (
  | {
      success: false;
      error: ITransactionError;
    }
  | {
      success: true;
      query?: IQuery;
      info?: IInfo;
      timestamp?: number;
      base?: string;
      date?: string;
      rates?: IRates;
    }
);

export interface ISuccessTransaction {
  success: true;
  query?: IQuery;
  info?: IInfo;
  result?: number;
  timestamp?: number;
  base?: string;
  date?: string;
  rates?: IRates;
}

export interface ITransactionOptions {
  from: string;
  to: string;
  amount: number;
}
